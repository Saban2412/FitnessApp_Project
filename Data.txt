using AutoSkola.Models;
using Microsoft.EntityFrameworkCore;

namespace AutoSkola.Data;

public class ApplicationDbContext:DbContext
{
    public DbSet<Instruktor> Instruktori { get; set; } = null!;
    public DbSet<Vozilo> Vozila { get; set; } = null!;

    public ApplicationDbContext(DbContextOptions<ApplicationDbContext> options) : base(options){}

    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        base.OnModelCreating(modelBuilder);

        modelBuilder.Entity<Instruktor>(entity =>
        {
            entity.ToTable("Instruktori");
            entity.HasKey(i => i.Id);
            entity.HasIndex(i => i.BrojDozvole).IsUnique();
            entity.Property(i => i.BrojDozvole).IsRequired().HasMaxLength(10);
            entity.Property(i => i.Ime).IsRequired().HasMaxLength(20);
            entity.Property(i => i.Prezime).IsRequired().HasMaxLength(25);
        });
        modelBuilder.Entity<Vozilo>(entity =>
        {
            entity.ToTable("Vozila");
            entity.HasKey(v => v.Id);
            entity.Property(v => v.Registracija).IsRequired();
            entity.HasIndex(v => v.Registracija).IsUnique();
            entity.Property(v => v.Marka).IsRequired().HasMaxLength(20);
            entity.Property(v => v.Model).IsRequired().HasMaxLength(20);

            entity.HasOne(v => v.Instruktor)
                .WithMany(v=>v.Vozila)
                .HasForeignKey(v => v.InstruktorId)
                .OnDelete(DeleteBehavior.SetNull);
        });
    }
}





using AutoSkola.Data; // ili AutoSkola.Data, prilagodi svom namespace-u
using Microsoft.EntityFrameworkCore;
using Microsoft.EntityFrameworkCore.Design;
using Microsoft.Extensions.Configuration;

namespace AutoSkola.Data;

public class ApplicationDbContextFactory : IDesignTimeDbContextFactory<ApplicationDbContext>
{
    public ApplicationDbContext CreateDbContext(string[] args)
    {
        var configuration = new ConfigurationBuilder()
            .SetBasePath(Directory.GetCurrentDirectory())
            .AddJsonFile("appsettings.json", optional: false)
            .Build();

        var connectionString = configuration.GetConnectionString("DefaultConnection");

        var optionsBuilder = new DbContextOptionsBuilder<ApplicationDbContext>();
        optionsBuilder.UseSqlite(connectionString);

        return new ApplicationDbContext(optionsBuilder.Options);
    }
}

using AutoSkola.Data;
using AutoSkola.Models;
using AutoSkola.Repositories;
using AutoSkola.Services;
using Microsoft.EntityFrameworkCore;
using Microsoft.Extensions.Configuration;
using Microsoft.Extensions.DependencyInjection;

var configuration = new ConfigurationBuilder()
    .SetBasePath(Directory.GetCurrentDirectory())
    .AddJsonFile("appsettings.json", optional:false)
    .Build();

var services=new ServiceCollection();
var connectionString =configuration.GetConnectionString("DefaultConnection");
services.AddDbContext<ApplicationDbContext>(options =>options.UseSqlite(connectionString));

services.AddScoped<IInstruktorRepository,InstruktorRepository>();
services.AddScoped<IVoziloRepository,VoziloRepository>();
services.AddScoped<IAutoskolaService,AutoskolaService>();

var provider = services.BuildServiceProvider();

var db=provider.GetRequiredService<ApplicationDbContext>();
await db.Database.EnsureCreatedAsync();

var app = provider.GetRequiredService<IAutoskolaService>();
